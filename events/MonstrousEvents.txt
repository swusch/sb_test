#Setup monstrous opinion
#Event effect doesn't work within a scripted effect
#Use this event after the 'clear_monstrous_opinion_effect = yes' script to refresh monstrous opinions for a country
country_event = {
	id = monstrous.1
	title = monstrous.1.t
	desc = monstrous.1.d
	picture = REFORM_eventPicture
	
	is_triggered_only = yes
	hidden = yes
	
	trigger = {
		NOT = { has_country_modifier = setup_monstrous_opinion }
	}
	
	immediate = {
		hidden_effect = {
			add_country_modifier = {
				name = setup_monstrous_opinion
				duration = -1
				hidden = yes
			}
		}
		# must be a modifier and not a flag so it disappears if the nation is annexed and released again
	}
	
	#Do it
	option = {
		name = monstrous.1.a
		ai_chance = { factor = 10 }
		if = {
			limit = {
				has_country_modifier = monstrous_nation
				is_colonial_nation = yes
				colonial_parent = { NOT = { has_country_modifier = monstrous_nation } }
			}
			set_country_flag = no_longer_monstrous
			remove_country_modifier = monstrous_nation
		}
		if = {
			limit = { has_country_modifier = monstrous_nation }
			# set opinions of the new monstrous nation
			every_country = {
				limit = { NOT = { tag = ROOT } } # do not set against yourself or error log is sad
				if = {
					limit = { has_country_modifier = monstrous_nation }
					add_opinion = { who = ROOT modifier = both_monstrous }
					reverse_add_opinion = { who = ROOT modifier = both_monstrous }
				}
				else_if = {
					limit = { ROOT = { has_country_flag = semi_monstrous } }
					add_opinion = { who = ROOT modifier = semi_monstrous }
					reverse_add_opinion = { who = ROOT modifier = semi_monstrous }
				}
				else = {
					add_opinion = { who = ROOT modifier = root_monstrous }
					reverse_add_opinion = { who = ROOT modifier = root_monstrous }
				}
			}
		}
		else = {
			# set opinions against other monstrous nations
			every_country = {
				limit = { 
					NOT = { tag = ROOT } # do not set against yourself or error log is sad
					has_country_modifier = monstrous_nation
				}
				if = {
					limit = { has_country_flag = semi_monstrous }
					add_opinion = { who = ROOT modifier = semi_monstrous }
					reverse_add_opinion = { who = ROOT modifier = semi_monstrous }
				}
				else = {
					add_opinion = { who = ROOT modifier = root_monstrous }
					reverse_add_opinion = { who = ROOT modifier = root_monstrous }
				}
			}
		}
	}
}

#Monstrous Opinion Update when culture swap
country_event = {
	id = monstrous.2
	title = monstrous.2.t
	desc = monstrous.2.d
	picture = ADVISOR_eventPicture
	
	trigger = {
		always = yes
	}
	
	hidden = yes
	is_triggered_only = yes
	
	option = {
		name = monstrous.2.a
		ai_chance = { factor = 100 }
		if = {
			limit = {
				OR = {
					AND = {
						NOT = { monstrous_culture = yes }
						has_country_modifier = monstrous_nation
					}
					AND = {
						monstrous_culture = yes
						is_subject = yes
						OR = {
							is_colonial_nation = yes
							overlord = { primary_culture = ROOT }
						}
						overlord = { NOT = { has_country_modifier = monstrous_nation } }
					}
				}
			}
			remove_country_modifier = monstrous_nation
			clr_country_flag = removing_monstrous
			if = { limit = { is_subject = yes } set_country_flag = no_longer_monstrous clr_country_flag = semi_monstrous }
			clear_monstrous_opinion_effect = yes
			country_event = { id = monstrous.1 } #opinion setup
		}
		else_if = {
			limit = {
				monstrous_culture = yes
				NOT = { has_country_modifier = monstrous_nation }
				NOT = { has_country_flag = no_longer_monstrous }
			}
			add_country_modifier = { name = monstrous_nation duration = -1 }
			if = {
				limit = {
					is_subject = yes
					OR = {
						is_colonial_nation = yes
						overlord = { primary_culture = ROOT }
					}
					overlord = { has_country_flag = semi_monstrous }
				}
				set_country_flag = semi_monstrous
			}
			clear_monstrous_opinion_effect = yes
			country_event = { id = monstrous.1 } #opinion setup
		}
	}
}
