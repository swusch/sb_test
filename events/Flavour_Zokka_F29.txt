	id = flavour_zokka.501
	title = flavour_zokka.501.t
	desc = flavour_zokka.501.d
	picture = SIEGE_eventPicture
	
	is_triggered_only = yes

	option = {
		name = flavour_zokka.501.a
		random_owned_province = {
			limit = { OR = { base_tax = 5 base_production = 5 base_manpower = 5 } is_capital = no }
			random_list = {
				1 = { 
					trigger = { base_tax = 5 } 
					add_base_tax = -4
					if = {
						limit = { ROOT = { has_reform = zokka_mouth_eats } }
						owner = { add_adm_power = 120 }
					}
					else = { owner = { add_adm_power = 240 } }
				}
				1 = { 
					trigger = { base_production = 5 } 
					add_base_production = -4
					if = {
						limit = { ROOT = { has_reform = zokka_mouth_eats } }
						owner = { add_dip_power = 120 }
					}
					else = { owner = { add_dip_power = 240 } }
				}
				1 = { 
					trigger = { base_manpower = 5 } 
					add_base_manpower = -4
					if = {
						limit = { ROOT = { has_reform = zokka_mouth_eats } }
						owner = { add_mil_power = 120 }
					}
					else = { owner = { add_mil_power = 240 } }
				}
			}
		}
	}
}

	id = flavour_zokka.1001

	desc = flavour_zokka.1001.d
	picture = BLANK_eventPicture
	
	is_triggered_only = yes
	trigger = { has_country_flag = zokka_something_is_coming_flag }
	hidden = yes

	option = {
		name = flavour_zokka.1001.a
		every_province = {
			limit = { has_province_modifier = zokka_trembling_airs }

				limit = { NOT = { has_province_modifier = zokka_trembling_airs } NOT = { owned_by = ROOT } }
				add_permanent_province_modifier = {
					name = zokka_trembling_airs

				}
			}
		}
	}
}

#Spreads Harimari
country_event = {
	id = flavour_zokka.1002
	title = flavour_zokka.1002.t
	desc = flavour_zokka.1002.d
	picture = BLANK_eventPicture
	
	is_triggered_only = yes
	trigger = { has_country_flag = zokka_maelstrom_flag }
	hidden = yes

	option = {
		name = flavour_zokka.1002.a
		every_owned_province = {
			limit = { culture = primal_harimari }
			every_neighbor_province = {
				limit = { 
					owned_by = ROOT 
					NOT = { culture_group = gnollish } 
					NOT = { culture = primal_harimari } 
					OR = { 
						superregion = rahen_superregion 
						superregion = vimdatrong_superregion 
						superregion = yanshen_superregion 
					} 
				}
				add_devastation = 30
				change_culture = primal_harimari
			}
		}
	}
}
