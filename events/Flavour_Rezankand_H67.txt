#Dispel the dark
country_event = {
	id = flavor_rezankand.6
	title = flavor_rezankand.6.t
	desc = flavor_rezankand.6.d
	picture = muslimgfx_PILGRIMAGE_eventPicture

	fire_only_once = yes
	is_triggered_only = yes

	option = {
		name = flavor_rezankand.6.a
		ai_chance = { factor = 100 }

		if = {
			limit = {
				OR = {
					check_variable = {
						which = rezankand_slave_provinces
						value = 7
					}
					AND = {
						has_country_flag = rezankand_not_enough_slaves
						num_of_owned_provinces_with = {
							value = 7
							religion = oren_nayiru
						}
					}
				}
			}
			custom_tooltip = rezankand_dispel_the_dark_bonus_tt
			nsc_update_incident_value = {
				increase = yes
				incident = incident_trapped_in_shadows
			}
			add_country_modifier = {
				name = rezankand_dispelled_darkness_modifier
				duration = -1
			}
			every_country = {
				limit = {
					capital_scope = {
						superregion = taychend_superregion
					}
					NOT = { religion = oren_nayiru }
				}
				add_country_modifier = {
					name = rezankand_dispel_the_dark_modifier
					duration = 36500 #100 years
				}
			}
			custom_tooltip = rezankand_dispel_the_dark_religion_bonus_tt
			hidden_effect = {
				change_variable = { #Bonus 20 dev or 4 more colonizable provinces
					which = rezankand_slave_provinces
					value = 2
				}
				taychend_superregion = {
					limit = {
						previous_owner = ROOT
					}
					random = {
						chance = 50
						change_religion = oren_nayiru
					}
				}
			}
		}
		#10 bonus dev or 2 frontiers per slave province
		if = {
			limit = {
				num_of_owned_provinces_with = {
					value = 10
					region = dry_coast_region
					is_city = yes
				}
			}
			custom_tooltip = rezankand_dispel_the_dark_dev_bonus_tt
			if = {
				limit = {
					OR = {
						check_variable = {
							which = rezankand_slave_provinces
							value = 7
						}
						num_of_owned_provinces_with = {
							value = 7
							religion = oren_nayiru
						}
					}
				}
				custom_tooltip = rezankand_dispel_the_dark_extra_dev_bonus_tt
			}
			hidden_effect = {
				while = {
					limit = {
						check_variable = {
							which = rezankand_slave_provinces
							value = 1
						}
					}
					for = {
						amount = 10
						effect = "
						random_owned_province = {
							limit = {
								region = dry_coast_region
								owned_by = ROOT
							}
							random_list = {
								1 = { add_base_tax = 1 }
								1 = { add_base_production = 1 }
								1 = { add_base_manpower = 1 }
							}
						}"
					}
					change_variable = {
						which = rezankand_slave_provinces
						value = -1
					}
				}
			}
		}
		else = {
			custom_tooltip = rezankand_dispel_the_dark_colony_bonus_tt
			if = {
				limit = {
					OR = {
						check_variable = {
							which = rezankand_slave_provinces
							value = 7
						}
						num_of_owned_provinces_with = {
							value = 7
							religion = oren_nayiru
						}
					}
				}
				custom_tooltip = rezankand_dispel_the_dark_extra_colony_bonus_tt
			}
			while = {
				limit = {
					check_variable = {
						which = rezankand_slave_provinces
						value = 1
					}
				}
				hidden_effect = {
					random_owned_province = {
						limit = {
							region = dry_coast_region
						}
						random_empty_neighbor_province = {
							add_siberian_construction = 700
							change_culture = vaikazelhi
						}
					}
					random_owned_province = {
						limit = {
							region = dry_coast_region
						}
						random_empty_neighbor_province = {
							add_siberian_construction = 700
							change_culture = vaikazelhi
						}
					}
					change_variable = {
						which = rezankand_slave_provinces
						value = -1
					}
				}
			}
		}
	}
}
