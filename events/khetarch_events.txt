namespace = khetarch_events

#Khetarch Amiskhet III  (1498-1532)
country_event = {
    id = khetarch_events.1
    title = khetarch_events.1.t
    desc = khetarch_events.1.d
    picture = ADVISOR_eventPicture

    trigger = {
        OR = {
            religion = elikhetist 
            religion = nirakhetist
        }
        is_year = 1498
        is_month = 4
        NOT = { is_year = 1532 }
        current_guru_is = khetarch_baqtkhet_ii
        NOT = { has_global_flag = guru_event_happening }
    }

    mean_time_to_happen = {
        days = 7
    }

    immediate = {
        set_global_flag = guru_event_happening
    }

    after = {
        clr_global_flag = guru_event_happening
    }

    option = {
        name = khetarch_events.1.a 
        appoint_next_guru = yes
    }
}
