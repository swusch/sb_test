#The Crown of Deshak
country_event = {
	id = flavour_khasa.341
	title = flavour_khasa.341.t
	desc = flavour_khasa.341.d
	picture = GOOD_WITH_MONARCH_eventPicture

	is_triggered_only = yes

	trigger = {
	}

	immediate = {
		hidden_effect = {
			U06 = {
				if = {
					limit = {
						is_subject = yes
					}
					overlord = { save_event_target_as = deshak_overlord }
				}
			}
		}
	}

	option = {
		name = flavour_khasa.341.a
		if = {
			limit = {
				U06 = { exists = yes }
			}
			create_subject = {
				subject_type = personal_union
				subject = U06
			}
			if = {
				limit = {
					has_country_flag = khasa_from_deshak
				}
				add_historical_friend = U06
				U06 = {
					add_historical_friend = ROOT
				}
			}
			if = {
				limit = {
					has_saved_event_target = deshak_overlord
				}
				event_target:deshak_overlord = {
					declare_war_with_cb = {
						who = U06
						casus_belli = cb_restore_personal_union
					}
				}
			}
		}
		else = {
			deshak_area = {
				add_nationalism = -10
			}
			oris_area = {
				add_nationalism = -10
			}
			if = {
				limit = {
					has_country_flag = khasa_from_deshak
				}
				add_stability_or_adm_power = yes
			}
		}
		ai_chance = {
			factor = 100
		}
	}
}
