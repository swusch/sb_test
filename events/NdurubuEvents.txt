namespace = ndurubu_events

#unification for Atijoko
country_event = {
	id = ndurubu_events.1
	title = ndurubu_events.1.t
	desc = ndurubu_events.1.d
	picture = ADVISOR_eventPicture

	is_triggered_only = yes
	fire_only_once = yes
	
	option = {
		name = ndurubu_events.1.a
		override_country_name = NDURUBU_NAME
		custom_tooltip = ndurubu_cores_tt
		hidden_effect = {
			fangaula_superregion = {
				limit = { culture = ndurubu }
				remove_core = S95
				remove_core = S94
				add_core = L16
			}
		}
		if  = {
			limit = { has_ruler_flag = atijoko_bwana_flag }
			add_ruler_modifier = { 
				name = L16_first_among_the_lions
				duration = -1  
			}
		}
	}
	
	option = {
		trigger = { 
			ai = no
		}
		name = ndurubu_events.1.b
		custom_tooltip = ndurubu_cores_tt
		hidden_effect = {
			fangaula_superregion = {
				limit = { culture = ndurubu }
				remove_core = S95
				remove_core = S94
				add_core = L16
			}
		}
		if  = {
			limit = { has_ruler_flag = atijoko_bwana_flag }
			add_ruler_modifier = { 
				name = L16_first_among_the_lions
				duration = -1  
			}
		}
	}
}
