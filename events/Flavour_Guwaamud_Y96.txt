country_event = { #Guwaadun Culture
	id = guwaamud.1
	title = guwaamud.1.t
	picture = GOOD_WITH_MONARCH_eventPicture
	desc = guwaamud.1.d
	
	is_triggered_only = yes

	option = { #Adopt the native ways
		name = guwaamud.1.a
		ai_chance = { factor = 100 }

		primary_and_ruler_culture_change = { race = human new_culture = guwaadun }
		every_owned_province = {
			limit = {
				OR = { 
					culture = shuvuush
					culture = kuuldaar
				} 
			}
			change_culture = guwaadun
		}
	}

	option = { #Hold true to the old ways
		name = guwaamud.1.b
		ai_chance = { factor = 0 }
		trigger = { NOT = { primary_culture = guwaadun } }

		anb_add_accepted_culture_or_dip_power = { culture = guwaadun free = yes }
		set_capital = 5348
		5348 = {
			move_capital_effect = yes
			change_culture = shuvuush
			change_province_name = "Bhengutai"
		}
	}
}

country_event = { #Manjang Shrine
	id = guwaamud.3
	title = guwaamud.3.t
	picture = GREAT_BUILDING_eventPicture
	desc = guwaamud.3.d

	is_triggered_only = yes

	option = { #Welp
		name = guwaamud.3.a
		ai_chance = { factor = 100 }
		if = {
			limit = {
				NOT = { stability = 3 }
			}
			add_stability = 1
			add_prestige = 5
		}
		else = {
			add_prestige = 15
		}
		add_legitimacy = 5
		5419 = {
			add_permanent_province_modifier = {
				name = Y96_somyo_temple
				duration = -1
			}
			change_religion = saanorgegh
		}
		if = {
			limit = { primary_culture = guwaadun }
			5419 = { change_culture = guwaadun }
		}
	}
}
