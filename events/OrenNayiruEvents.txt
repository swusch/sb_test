# Oren Nayiru Intro/Warning Event
country_event = {
	id = oren_nayiru.1
	title = oren_nayiru.1.t #First Contact
	desc = oren_nayiru.1.d
	picture = TREASURE_FLEET_SHIP_eventPicture
	
	fire_only_once = yes
	trigger = {
		NOT = {
			OR = {
				has_global_flag = oren_nayiru_intro
				has_global_flag = oren_nayiru_intro_early
			}
		}
		OR = {
			exists = H66 #Jaherian Exemplars
			exists = H67 #Rezankand
			is_year = 1550
		}
		any_owned_province = { superregion = taychend_superregion }
	}

	mean_time_to_happen = { days = 1 }
	
	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					any_owned_province = { superregion = taychend_superregion }
					NOT = { tag = ROOT }
				}
				country_event = { id = oren_nayiru.910 }
			}
			if = {
				limit = { 
					is_year = 1500
					NOT = {
						J78 = { has_country_flag = uesrennu_early_oren }
					}
				}
				set_global_flag = oren_nayiru_intro
			}
			else = {
				set_global_flag = oren_nayiru_intro_early # prevents oren nayiru from spawning in the 1400s (too early)
			}
		}
	}
	
	option = {
		name = oren_nayiru.1.a
		if = {
			limit = { is_or_was_tag = { tag = J78 } }
			custom_tooltip = uesrennu_warning
		}
	}
}

country_event = {
	id = oren_nayiru.4
	title = oren_nayiru.4.t #The Calsithara Revolt
	desc = {
		trigger = { NOT = { is_or_was_tag = { tag = J78 } } }
		desc = oren_nayiru.4.d
	}
	desc = { 
		trigger = { 
			is_or_was_tag = { tag = J78 }
			NOT = { has_country_flag = uesrennu_rat_emperor }
		}
		desc = oren_nayiru.4.d1	#uesrennu
	}
	desc = { 
		trigger = { has_country_flag = uesrennu_rat_emperor }
		desc = oren_nayiru.4.d2	#ditto
	}
	picture = wihgfx_FEMALE_GENERAL_eventPicture
	
	fire_only_once = yes
	is_triggered_only = yes
	
	trigger = {
		has_disaster_progress = {
			disaster = kamrayakvals_war
			value = 50
		}
	}
	
	immediate = {
		hidden_effect = {
			every_country = {
				limit = {
					NOT = { tag = ROOT }
					has_disaster_progress = {
						disaster = kamrayakvals_war
						value = 1
					}
				}
				country_event = {
					id = oren_nayiru.940
				}
			}
		}
	}
	
	option = {
		name = oren_nayiru.4.a
		taychend_superregion = {
			limit = { 
				religion = oren_nayiru
				is_reformation_center = no
                owner = { NOT = { religion = oren_nayiru } }
				NOT = { province_id = 2502 } #Uesrennu
			}
			if = {
				limit = { 2502 = { check_variable = { which = onVariable value = 2 } } }
				oren_nayiru_rebels = 1
				remove_province_modifier = religious_zeal_at_conv
				set_province_flag = calsithara_revolt_province_flag
				hidden_effect = { 2502 = { set_variable = { which = onVariable value = 0 } } }
			}
			else = {
				hidden_effect = { 2502 = { change_variable = { which = onVariable value = 1 } } }
			}
		}
		if = {
			limit = { is_or_was_tag = { tag = J78 } } #Uesrennu
			if = {
				limit = { has_heir = yes }
				kill_heir = {}
			}
			2502 = { add_devastation = 25 } #Uesrennu
			if = { 
				limit = { NOT = { has_country_flag = uesrennu_rat_emperor } }
				define_ruler = { dynasty = "Uesrenna" }
				hidden_effect = { set_dynasty = "Uesrenna" }
				custom_tooltip = uesrennu_dream_is_dead_tt
				clr_country_flag = rat_kings_alive
			}
		}
		if = {
			limit = {
				2502 = { owner = { NOT = { religion = oren_nayiru } } } #Uesrennu
			}
			2502 = { #Uesrennu
				spawn_rebels = {
					type = oren_nayiru_rebels
					size = 3
					leader = "Calsithara Flamespeaker"
				}
				set_province_flag = calsithara_revolt_leader_province_flag
			}
		}
		else = {
			random_province = {
				limit = {
					superregion = taychend_superregion
					religion = oren_nayiru
					owner = { NOT = { religion = oren_nayiru } }
					NOT = { has_province_flag = calsithara_revolt_province_flag }
				}
				spawn_rebels = {
					type = oren_nayiru_rebels
					size = 3
					leader = "Calsithara Flamespeaker"
				}
				set_province_flag = calsithara_revolt_leader_province_flag
			}
		}
	}
}
