country_event = {
    id = vaengheim.99
    title = vaengheim.99.t
    desc = vaengheim.99.d
    picture = COMBINED_HAGS_eventPicture
    is_triggered_only = yes
    hidden = yes
    trigger = {
        has_country_flag = Z88_reachmen_in_parliament_f #this is the first flag its possible to get from the below effects
        is_or_was_tag = {
            tag = Z88
        }
    }
    option = {
        name = vaengheim.28.a
        if = {
            limit = {
                has_country_flag = Z88_kobolds_in_parliament_f
            }
            random_owned_province = {
                limit = {
                    culture_group = kobold
                    NOT = {
                        culture = whitescale_kobold
                    }
                    OR = {
                        harpy_roost_st_has_any_roost = yes
                        any_neighbor_province = {
                            OR = {
                                culture = whitescale_kobold
                                harpy_roost_st_has_any_roost = yes
                            }
                        }
                    }
                }
                change_culture = whitescale_kobold
            }
        }
        if = {
            limit = {
                has_country_flag = Z88_gnomes_in_parliament_f
            }
            random_owned_province = {
                limit = {
                    culture_group = gnomish
                    NOT = {
                        culture = glacier_gnome
                    }
                    OR = {
                        harpy_roost_st_has_any_roost = yes
                        any_neighbor_province = {
                            OR = {
                                culture = glacier_gnome
                                harpy_roost_st_has_any_roost = yes
                            }
                        }
                    }
                    
                }
                change_culture = glacier_gnome
            }
        }
        if = {
            limit = {
                has_country_flag = Z88_reachmen_in_parliament_f
            }

                limit = {
                    OR = {
                        culture = blue_reachman

            limit = {
                has_country_flag = Z88_skamvin_in_parliament_f
            }
            random_owned_province = {
                limit = {
                    culture_group = alenic
                    NOT = {

                    }

                                culture = skamvin
