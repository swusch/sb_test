	title = flavour_brrtekuh.999.t
	desc = flavour_brrtekuh.999.d
	picture = muslimgfx_DEBATE_REPUBLICAN_eventPicture
    is_triggered_only = yes
    option = {
		name = flavour_brrtekuh.999.a

			limit = { NOT = { is_subject_of_type = personal_union } }
			change_government = republic

		}

		}

			add_country_modifier = {

#Gnoll in the High Tower
country_event = {
    id = flavour_brrtekuh.2
	title = flavour_brrtekuh.2.t
	desc = flavour_brrtekuh.2.d
	picture = OGRE_FEAST_eventPicture
    is_triggered_only = yes
    option = {
		name = flavour_brrtekuh.2.a
        A47 = {
			if = {
				limit = { has_heir = yes }
				kill_heir = { }
			}
			if = {
				limit = { has_regency = no }
				kill_ruler = yes
			}
			if = {
				limit = { has_consort = yes }
				remove_consort = yes
			}
			release_all_subjects = yes
        }
		deshak_area = { add_permanent_claim = ROOT }
		oris_area = { add_permanent_claim = ROOT }
		every_war_enemy_country = {
			limit = {
				is_in_war = {
					attacker_leader = ROOT
					defenders = A47
				}
				NOT = {
					tag = U06
					tag = A47
				}
			}
			white_peace = ROOT
		}
		if = {
			limit = { has_ruler_flag = THE_brrtekuh }
			hidden_effect = {
				change_variable = {
					which = ate_rulers
					value = 1
				}
			}
			add_ruler_modifier = {
				name = U10_feller_of_the_towers
				duration = -1
			}
		}
		else = { add_dip_power = 50 }
    }
}
