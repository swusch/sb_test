country_decisions = {

	rinta_invest_in_telescope_decision = {
		potential = {
			is_incident_active = incident_rinta_light_up_the_night_sky
			has_country_flag = enabled_light_up_the_night_sky_decision
		}
		allow = {
			is_incident_active = incident_rinta_light_up_the_night_sky
			hidden_trigger = { has_country_flag = enabled_light_up_the_night_sky_decision }
			treasury = 100
		}
		effect = {
			country_event = {
				id = rinta_seeker_events.599 #Invest In Telescope Event
			}
		}
		ai_will_do = {
			factor = 1
		}
	}

	rinta_steal_telescope_funds = {
		potential = {
			is_incident_active = incident_rinta_light_up_the_night_sky
			has_country_flag = enabled_light_up_the_night_sky_decision
			NOT = { has_global_flag = rinta_disable_stealing_funds_flag }
		}
		provinces_to_highlight = {
			is_capital = yes
			owner = {
				is_incident_active = incident_rinta_light_up_the_night_sky
				OR = {
					AND = {
						has_global_flag = rinta_supercharge_stealing_funds_flag
						NOT = { 
							has_spy_network_from = {
								who = ROOT
								value = 20
							}
						}
					}
					AND = {
						NOT = { 
							has_global_flag = rinta_supercharge_stealing_funds_flag
							has_spy_network_from = {
								who = ROOT
								value = 35
							}
						}
					}
				}
			}
		}
		allow = {
			is_incident_active = incident_rinta_light_up_the_night_sky
			hidden_trigger = { 
				has_country_flag = enabled_light_up_the_night_sky_decision 
				NOT = { has_global_flag = rinta_disable_stealing_funds_flag }
			}
			any_known_country = {
				is_incident_active = incident_rinta_light_up_the_night_sky
				if = {
					limit = { has_global_flag = rinta_supercharge_stealing_funds_flag }
					has_spy_network_from = {
						who = ROOT
						value = 20
					}
				}
				else = {
					has_spy_network_from = {
						who = ROOT
						value = 35
					}
				}
			}
			if = {
				limit = { has_country_flag = stole_telescope_funds_flag }
				had_country_flag = {
					flag = stole_telescope_funds_flag
					days = 365 #1 year
				}
			}
		}
		effect = {
			country_event = {
				id = rinta_seeker_events.598 #Steal Funds Event
			}
		}
		ai_will_do = {
			factor = 1
		}
	}

	rinta_raise_foreign_funds = {
		potential = {
			is_incident_active = incident_rinta_light_up_the_night_sky
			has_country_flag = enabled_rinta_raise_foreign_funds_decision
		}
		provinces_to_highlight = {
			is_capital = yes
			owner = {
				OR = {
					AND = {
						ROOT = { has_country_flag = limit_rinta_raise_foreign_funds_decision }
						religion_group = east_sarhal_religion
						has_opinion = {
							who = ROOT
							value = 50
						}
					}
					AND = {
						NOT = { ROOT = { has_country_flag = limit_rinta_raise_foreign_funds_decision } }
						has_opinion = {
							who = ROOT
							value = 25
						}
					}
				}
				NOT = { has_country_flag = rinta_invested_in_telescope_for_@ROOT }
			}
		}
		allow = {
			is_incident_active = incident_rinta_light_up_the_night_sky
			hidden_trigger = { has_country_flag = enabled_rinta_raise_foreign_funds_decision }
			any_known_country = {
				if = {
					limit = { ROOT = { has_country_flag = limit_rinta_raise_foreign_funds_decision } }
					religion_group = east_sarhal_religion
					has_opinion = {
						who = ROOT
						value = 50
					}
				}
				else = {
					has_opinion = {
						who = ROOT
						value = 25
					}
				}
				custom_trigger_tooltip = {
					tooltip = rinta_country_hasnt_invested_tt
					NOT = { has_country_flag = rinta_invested_in_telescope_for_@ROOT }
				}
			}
		}
		effect = {
			if = {
				limit = { has_country_flag = limit_rinta_raise_foreign_funds_decision }
				every_known_country = {
					limit = {
						religion_group = east_sarhal_religion
						has_opinion = {
							who = ROOT
							value = 50
						}
						NOT = { has_country_flag = rinta_invested_in_telescope_for_@ROOT }
					}
					ROOT = { rinta_invest_in_telescope = { level = 0_5 } }
					add_opinion = {
						who = ROOT
						modifier = rinta_invested_in_telescope
					}
					set_country_flag = rinta_invested_in_telescope_for_@ROOT
				}
			}
			else = {
				every_known_country = {
					limit = {
						has_opinion = {
							who = ROOT
							value = 25
						}
						NOT = { has_country_flag = rinta_invested_in_telescope_for_@ROOT }
					}
					ROOT = { rinta_invest_in_telescope = { level = 0_25 } }
					set_country_flag = rinta_invested_in_telescope_for_@ROOT
				}
			}
		}
		ai_will_do = {
			factor = 1
		}
	}
}
