lose_ia = {
	trigger = {
		has_reached_seat_threshold = no
		is_emperor = yes
		OR = {
			imperial_influence = 14
			AND = {
				imperial_influence = 4
				NOT = {
					# these debates cost 10 imperial authority on their own
					current_debate = consolidate_courts
					current_debate = abolish_imperial_borders
					current_debate = anbennar_de_jure_law
					current_debate = edict_of_peace
					current_debate = edict_enforcing_religion
					current_debate = anbennar_unification_sentiment
				}
			}
		}
		has_not_bribe_blocking_debate_active = {
			type = imperial_authority
		}
	}
	
	effect = {
		if = {
			limit = {
				has_expensive_debate = yes
				NOT = { halved_parliament_bribes = yes }
			}
			add_imperial_influence = -4
		}
		else_if = {
			limit = {
				OR = {
					AND = {
						has_expensive_debate = yes
						halved_parliament_bribes = yes
					}
					AND = {
						NOT = { has_expensive_debate = yes }
						NOT = { halved_parliament_bribes = yes }
					}
				}
			}
			add_imperial_influence = -2
		}
		else = {
			add_imperial_influence = -1
		}
	}
	
	chance = {
		factor = 1
	}

	ai_will_do = {
		factor = 1
	}
}

national_lose_ia = {
	trigger = {
		has_reached_seat_threshold = yes
		is_emperor = yes
		OR = {
			imperial_influence = 19
			AND = {
				imperial_influence = 9
				NOT = {
					# these debates cost 10 imperial authority on their own
					current_debate = consolidate_courts
					current_debate = abolish_imperial_borders
					current_debate = anbennar_de_jure_law
					current_debate = edict_of_peace
					current_debate = edict_enforcing_religion
					current_debate = anbennar_unification_sentiment
				}
			}
		}
		has_not_bribe_blocking_debate_active = {
			type = imperial_authority
		}
	}
	
	effect = {
		back_seats_of_bribe_type = {
			bribe = lose_ia
		}
		if = {
			limit = {
				has_expensive_debate = yes
				NOT = { halved_parliament_bribes = yes }
			}
			add_imperial_influence = -9
		}
		else_if = {
			limit = {
				OR = {
					AND = {
						has_expensive_debate = yes
						halved_parliament_bribes = yes
					}
					AND = {
						NOT = { has_expensive_debate = yes }
						NOT = { halved_parliament_bribes = yes }
					}
				}
			}
			add_imperial_influence = -6
		}
		else = {
			add_imperial_influence = -3
		}
	}
	
	ai_will_do = {
		factor = 1
	}
}
