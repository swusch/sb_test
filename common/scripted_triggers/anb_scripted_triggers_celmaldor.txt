is_giesadh = {
	custom_trigger_tooltip = {
		tooltip = z22_is_giesadh_tt
		OR = {
			has_country_flag = z22_tag_is_giesadh
			has_country_flag = z22_is_permanent_giesadh
		}
	}
}

is_giesadh_or_vic_league_member = {
	custom_trigger_tooltip = {
		tooltip = z22_is_giesadh_or_vic_league_member_tt
		OR = {
			has_country_flag = z22_tag_is_giesadh
			has_country_flag = z22_is_permanent_giesadh
			is_subject_of_type_with_overlord = {
				who = Z22
				type = vic_league_member
			}
			AND = {
				B98 = { was_tag = Z22 }
				is_subject_of_type_with_overlord = {
					who = B98
					type = vic_league_member
				}
			}
		}
	}
}

is_NOT_giesadh_or_vic_league_member = {
	custom_trigger_tooltip = {
		tooltip = z22_is_NOT_giesadh_or_vic_league_member_tt
		NOT = {
			has_country_flag = z22_tag_is_giesadh
			has_country_flag = z22_is_permanent_giesadh
			is_subject_of_type_with_overlord = {
				who = Z22
				type = vic_league_member
			}
			AND = {
				B98 = { was_tag = Z22 }
				is_subject_of_type_with_overlord = {
					who = B98
					type = vic_league_member
				}
			}
		}
	}
}

is_former_colony_or_ally_in_aelantir = {
	custom_trigger_tooltip = {
		tooltip = z22_is_former_colony_or_ally_in_aelantir_tt
		OR = {
			has_country_flag = z22_former_celmaldor_colonial_tag
			AND = {
				OR = {
					alliance_with = Z22
					AND = {
						B98 = {
							was_tag = Z22
						}
						alliance_with = B98
					}
				}
				capital_scope = {
					OR = {
						continent = north_america
						continent = south_america
					}
				}
			}
		}
	}
}
