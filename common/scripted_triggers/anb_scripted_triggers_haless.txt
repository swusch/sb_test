character_has_personality_or_ancestor = {
	if = {
		limit = { religion = neokttareum }
		$character$_has_personality = ancestor_$personality$
	}
	else = { $character$_has_personality = $personality$ }
}
ruler_has_personality_or_ancestor = {
	character_has_personality_or_ancestor = { character = ruler personality = $personality$ }
}
heir_has_personality_or_ancestor = {
	character_has_personality_or_ancestor = { character = heir personality = $personality$ }
}
queen_has_personality_or_ancestor = { #Aliasing both
	character_has_personality_or_ancestor = { character = queen personality = $personality$ }
}
consort_has_personality_or_ancestor = {
	character_has_personality_or_ancestor = { character = queen personality = $personality$ }
}

province_has_facestealer_trigger = {
	custom_trigger_tooltip = {
		tooltip = province_has_facestealer_tt
		has_province_flag = province_has_facestealer
	}
}
province_has_no_facestealer_trigger = {
	custom_trigger_tooltip = {
		tooltip = province_has_no_facestealer_tt
		NOT = { has_province_flag = province_has_facestealer }
	}
}

can_use_mindarayan = {
	if = { #Hisost Yamok
		limit = { owner = { has_country_flag = s23_bigmount } }
		custom_trigger_tooltip = {
			tooltip = s23_bigmount_t
			always = yes
		}
	}
	else_if = { #Yuanszi
		limit = { has_province_flag = wonder_religion_bypass_flag }
		has_owner_religion = yes
	}
	else_if = { #Zokka
		limit = { owner = { tag = F29 } }
		custom_trigger_tooltip = {
			tooltip = zokka_minda_usage_tt
			owner = { has_country_flag = zokka_minda_flag }
		}
	}
	else_if = { #Azkare/sunrise empire
		limit = { owner = { is_or_was_tag = { tag = Y86 } } }
		custom_trigger_tooltip = {
			tooltip = azkare_oracle_mountain_tt
			owner = { 
				OR = {
					religion = accretive_path
					religion = transmutative_path
					religion = mystic_accord
				}
			}
		}
		custom_trigger_tooltip = {
			tooltip = azkare_tolerance_monuments_tt
			tolerance_to_this = 1
		}
	}
	else = {
		OR = {
			religion_group = raheni
			religion = accretive_path
			religion = transmutative_path
			religion = mystic_accord
			religion = neokttareum
			religion = saanorgegh
		}
		has_owner_religion = yes
	}
}
