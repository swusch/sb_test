konolkhatep_slave_project_trigger_yes = {
	has_government_power = {
		mechanic_type = konolkhatep_mechanic
		power_type = konolkhatep_slaves
		value = $slaves$
	}
	owns_or_subject_of = $province$
	NOT = { has_country_flag = konolkhatep_slave_project_ongoing }
	NOT = { has_country_flag = konolkhatep_$project$_flag }
}

		NOT = { owns_or_subject_of = $province$ }
	}
	NOT = { has_country_flag = konolkhatep_$project$_flag }
}

konolkhatep_convert_satrapy_trigger_yes = {
	has_saved_event_target = konol_satrapy_target_$target$
	check_variable = {
		which = konol_available_dip
		which = konol_satrapy_variable_$target$
	}
	OR = {
		AND = {
			event_target:konol_satrapy_target_$target$ = { total_development = 500 }
			has_government_power = {
				mechanic_type = konolkhatep_mechanic
				power_type = konolkhatep_slaves
				value = 360
			}
		}
		AND = {
			event_target:konol_satrapy_target_$target$ = { total_development = 250 NOT = { total_development = 500 } }
			has_government_power = {
				mechanic_type = konolkhatep_mechanic
				power_type = konolkhatep_slaves
				value = 180
			}
		}
		event_target:konol_satrapy_target_$target$ = { NOT = { total_development = 250 } }
	}
}

konolkhatep_convert_satrapy_trigger_no = {
	has_saved_event_target = konol_satrapy_target_$target$
	OR = {
		NOT = {
			check_variable = {
				which = konol_available_dip
				which = konol_satrapy_variable_$target$
			}
		}
		AND = {
			event_target:konol_satrapy_target_$target$ = { total_development = 500 }
			NOT = {
				has_government_power = {
					mechanic_type = konolkhatep_mechanic
					power_type = konolkhatep_slaves
					value = 360
				}
			}
		}
		AND = {
			event_target:konol_satrapy_target_$target$ = { total_development = 250 NOT = { total_development = 500 } }
			NOT = {
				has_government_power = {
					mechanic_type = konolkhatep_mechanic
					power_type = konolkhatep_slaves
					value = 180
				}
			}
		}
	}
}
