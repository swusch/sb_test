
estate_mages_disaster = {
	potential = {
		has_estate = estate_mages
		estate_influence = {
			estate = estate_mages
			influence = 80
		}
		NOT = { has_country_flag = farranean_estates_harmony }
		NOT = {
			AND = {
				has_government_attribute = is_magocracy
				ruler_is_powerful_mage = yes
			}
		}
	}

	can_start = {
		has_any_disaster = no
		estate_influence = {
			estate = estate_mages
			influence = 100
		}
		ruler_is_not_witch_king = yes
		is_at_war = no
	}
	
	can_stop = {
		OR = {
			has_any_disaster = yes
			ruler_is_witch_king = yes
			has_government_attribute = is_magocracy
			NOT = {
				estate_influence = {
					estate = estate_mages
					influence = 100
				}		
			}
		}
	}
	
	down_progress = {
		factor = 0
		modifier = {
			factor = 1
			NOT = {
				estate_influence = {
					estate = estate_mages
					influence = 100
				}
			}
		}
	}
	
	progress = {
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_mages
					loyalty = 50
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 0.5
			NOT = {
				estate_loyalty = {
					estate = estate_mages
					loyalty = 40
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_mages
					loyalty = 30
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = 1
			NOT = {
				estate_loyalty = {
					estate = estate_mages
					loyalty = 20
				}
			}
			is_at_war = no
		}
		modifier = {
			factor = -1
			ruler_is_powerful_mage = yes
		}
	}
	
	can_end = {
		OR = {
			custom_trigger_tooltip = {
				tooltip = EST_CRUSHED_MAGES
				NOT = { has_country_flag = estate_mages_in_power }
			}
			has_government_attribute = is_magocracy
			custom_trigger_tooltip = {
				tooltip = EST_SUCCESS_MAGES
				had_country_flag = { flag = estate_mages_in_power days = 3650 }
			}
		}
	}
	
	modifier = {	
		global_tax_modifier = -0.5
		prestige = -0.5
	}

	on_start = estate_disasters.200
	on_end = estate_disasters.201
	
	on_monthly = {
	}
}
