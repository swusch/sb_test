							}
							AND = {
								has_terrain = city_terrain
								development = 20
							}
							development = 30
							# province_id = 2957 # Regensburg
							# province_id = 1868 # Augsburg
							# province_id = 1872 # Ulm
						}
					}
					province_event = {
						id = hre_event.6
					}
					add_province_triggered_modifier = perpetual_diet
				}
			}
		}
	}
	off_effect = {
		every_province = {
			remove_province_triggered_modifier = perpetual_diet
