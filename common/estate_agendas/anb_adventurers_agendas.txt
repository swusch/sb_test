

#War with monstrous nation
estate_adventurers_war_monstrous = {
	can_select = {
		is_free_or_tributary_trigger = yes
		is_at_war = no
		any_neighbor_country = {
			has_country_modifier = monstrous_nation
			is_free_or_tributary_trigger = yes
			NOT = { is_subject_of = root }
			NOT = { overlord_of = root }
			NOT = { alliance_with = root }
			NOT = { ROOT = { truce_with = prev } }
			NOT = { 
				has_opinion = {
					who = root
					value = 25
				}
			}
			ROOT = {
				knows_country = prev
			}
			OR = {
				is_part_of_hre = no
				NOT = { 
					OR = {
						hre_reform_passed = anbennar_mages_peace
						hre_reform_passed = anbennar_empower_imperial_diet
					}
				}
				ROOT = { is_part_of_hre = no }
			}
		}
	}
	selection_weight = {
		factor = 1
		modifier = {
			factor = 0.5
			has_non_generic_missions = yes
		}
		modifier = {
			factor = 1.5
			num_of_cities = 20
		}
		modifier = {
			factor = 1.5
			num_of_cities = 25
		}
		modifier = {
			factor = 1.5
			num_of_cities = 30
		}
		modifier = {
			factor = 1.5
			num_of_cities = 40
		}
		modifier = {
			factor = 1.5
			num_of_cities = 50
		}
		modifier = {
			factor = 0.25
			any_country = {
				coalition_target = root
			}
		}
	}
	pre_effect = {
		random_neighbor_country = {
			limit = {
				has_country_modifier = monstrous_nation
				is_free_or_tributary_trigger = yes
				NOT = { is_subject_of = root }
				NOT = { overlord_of = root }
				NOT = { alliance_with = root }
				NOT = { ROOT = { truce_with = prev } }
				NOT = { 
					has_opinion = {
						who = root
						value = 25
					}
				}
				ROOT = {
					knows_country = prev
				}
				OR = {
					is_part_of_hre = no
					NOT = { 
						OR = {
							hre_reform_passed = anbennar_mages_peace
							hre_reform_passed = anbennar_empower_imperial_diet
						}
					}
					ROOT = { is_part_of_hre = no }
				}
			}
			save_event_target_as = agenda_country
		}
	}
	provinces_to_highlight = {
		is_capital_of = event_target:agenda_country
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = { estate = estate_adventurers }
			}
			has_estate_agenda_auto_completion = { estate = estate_adventurers }
		}
		else = {
			event_target:agenda_country = {
				war_with = ROOT
			}
		}
	}
	fail_if = {
		event_target:agenda_country = {
			NOT = { num_of_cities = 1 }
		}
	}
	task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_adventurers
		}
		add_estate_loyalty = {
			estate = estate_adventurers
			loyalty = 10
		}
	}
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_adventurers
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
		}
    }
}

				}
			}

				knows_country = prev
			}
			OR = {
				is_part_of_hre = no
				NOT = { 
					OR = {
						hre_reform_passed = anbennar_mages_peace
						hre_reform_passed = anbennar_empower_imperial_diet
					}
				}
				ROOT = { is_part_of_hre = no }
			}
		}
	}
	selection_weight = {
		factor = 2
		modifier = {
			factor = 0.5
			has_non_generic_missions = yes
		}
		modifier = {
			factor = 1.5
			num_of_cities = 20
		}
		modifier = {
			factor = 1.5
			num_of_cities = 25
		}
		modifier = {
			factor = 1.5
			num_of_cities = 30
		}
		modifier = {
			factor = 1.5
			num_of_cities = 40
		}
		modifier = {
			factor = 1.5
			num_of_cities = 50
		}
		modifier = {
			factor = 0.25
			any_country = {
				coalition_target = root
			}
		}
	}
	pre_effect = {
		random_neighbor_country = {
			limit = {
				has_country_modifier = monstrous_nation
				num_of_cities = 6
				is_free_or_tributary_trigger = yes
				NOT = { is_subject_of = root }
				NOT = { overlord_of = root }
				NOT = { alliance_with = root }
				NOT = { ROOT = { truce_with = prev } }
				NOT = { 
					has_opinion = {
						who = root
						value = 25
					}
				}
				NOT = {
					any_owned_province = { #No need for this agenda if covered by a mission
						OR = {
							is_permanent_claim = root
							is_core = root
						}
					}
				}
				any_owned_province = { #any_neighbor_country includes neighboring tribal land, but any_neighbor_province doesn't - check we have an actual province bordering
					any_neighbor_province = {
						owned_by = ROOT
					}
				}
				ROOT = {
					knows_country = prev
				}
				OR = {
					is_part_of_hre = no
					NOT = { 
						OR = {
							hre_reform_passed = anbennar_mages_peace
							hre_reform_passed = anbennar_empower_imperial_diet
						}
					}
					ROOT = { is_part_of_hre = no }
				}
			}
			random_owned_province = {
				limit = {
					any_neighbor_province = {
						owned_by = root
					}
				}
				save_event_target_as = agenda_province
			}
		}
	}
	immediate_effect = {
		if = {
			limit = {
				event_target:agenda_province = {
					NOT = { is_claim = root }
					NOT = { is_core = root }
				}
			}
			event_target:agenda_province = {
				add_claim = root
			}
		}
	}
	on_invalid = {
		if = {
			limit = {
				event_target:agenda_province = {
					is_claim = root
					NOT = { is_permanent_claim = root }
				}
			}
			event_target:agenda_province = {
				remove_claim = root
			}
		}
	}
	provinces_to_highlight = {
		province_id = event_target:agenda_province
	}
	task_requirements = {
		if = {
			limit = {
				has_estate_agenda_auto_completion = { estate = estate_adventurers }
			}
			has_estate_agenda_auto_completion = { estate = estate_adventurers }
		}
		else = {
			event_target:agenda_province = {
				owned_by = root
			}
		}
	}
	fail_if = {
		event_target:agenda_province = {
			NOT = { is_claim = root }
			NOT = { is_core = root }
			NOT = { owned_by = root }
		}
	}
	task_completed_effect = {
		on_completed_agenda_effect = {
			estate = estate_adventurers
		}
		add_estate_loyalty = {
			estate = estate_adventurers
			loyalty = 5
		}
		add_adm_power = 50
	}
    failing_effect = {
		on_failed_agenda_effect = yes
		clr_auto_complete_flag = yes
        add_estate_loyalty_modifier = {
            estate = estate_adventurers
			desc = EST_VAL_AGENDA_DENIED
			loyalty = -5
			duration = 7300
        }
		if = {
			limit = {
				event_target:agenda_province = {
					is_claim = root
					NOT = { is_permanent_claim = root }
				}
			}
			event_target:agenda_province = {
				remove_claim = root
			}
		}
    }
}

#Colonise something in X area
estate_adventurers_colonise_X = {
	can_select = {
		num_of_colonists = 1
		num_of_ports = 1
		any_province = {
			range = root
			is_city = no
			is_wasteland = no
			NOT = { has_province_modifier = cant_colonize }
			OR = {
				has_port = yes
				any_neighbor_province = {
					OR = {
						owned_by = root
						owner = {
							is_colonial_nation_of = root
						}
					}
				}
			}
			has_discovered = root
			area_for_scope_province = {
				type = all
				NOT = { owned_by = root }
				NOT = { has_province_modifier = cant_colonize }
				NOT = {
					owner = {
						is_colonial_nation_of = root
					}
				}

						}
					}
