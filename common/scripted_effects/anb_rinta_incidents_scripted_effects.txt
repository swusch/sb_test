rinta_get_telescope_multiplier = {
	export_to_variable = {
		which = $variable$
		value = modifier:telescope_investment_efficency
		who = event_target:night_sky_country_target #Using ROOT instead doesn't work if anyone's curious
	}
	change_variable = {
		which = $variable$
		value = 1
	}
	[[value] #For flat amounts
		multiply_variable = {
			which = $variable$
			value = $value$
		}
	]
	[[which] #For dynamic amounts (everything else)
		multiply_variable = {
			which = $variable$
			which = $which$
		}
	]
}

rinta_reduce_investment_in_telescope = {

			subtract_variable = { #Update your total amount raised
				which = rinta_telescope_amount_invested
				value = $amount$
			}

			set_country_flag = rinta_telescope_triggered_amount #To remove the right cash
		}
	]
	[[years_of_income] #Invest a certain number of years of income
		custom_tooltip = rinta_reduce_investment_in_telescope_$years_of_income$_yoi_tt
		hidden_effect = {
			export_to_variable = {
				which = sky_incident_treasury_country_temp
				value = years_of_income
			}
			multiply_variable = { #Set the amount being invested to however many yoi are wanted
				which = sky_incident_treasury_country_temp
				value = $years_of_income$
			}
			subtract_variable = { #Update your total amount raised
				which = rinta_telescope_amount_invested
				which = sky_incident_treasury_country_temp
			}
		}
	]
	[[all] #Invest all of your country's money
		custom_tooltip = rinta_reduce_investment_in_telescope_all_tt
		hidden_effect = {
			export_to_variable = {
				which = sky_incident_treasury_country_temp
				value = treasury
			}
			subtract_variable = { #Update your total amount raised
				which = rinta_telescope_amount_invested
				which = sky_incident_treasury_country_temp
			}
		}
	]
	[[half] #Invest 1/2 your country's money
		custom_tooltip = rinta_reduce_investment_in_telescope_half_tt
		hidden_effect = {
			export_to_variable = {
				which = sky_incident_treasury_country_temp
				value = treasury
			}
			divide_variable = { #Set the amount being invested to half your treasury
				which = sky_incident_treasury_country_temp
				value = 2
			}
			subtract_variable = { #Update your total amount raised
				which = rinta_telescope_amount_invested
				which = sky_incident_treasury_country_temp
			}
		}
	]
	[[level] #Invest based on level of telescope
		custom_tooltip = rinta_reduce_investment_in_telescope_level_$level$_tt
		hidden_effect = {
			subtract_variable = { #Update your total amount raised
				which = rinta_telescope_amount_invested
				which = rinta_telescope_lvl_$level$_cost
			}
			set_variable = {
				which = sky_incident_treasury_country_temp
				which = rinta_telescope_lvl_$level$_cost
			}
		}
	]

	[[refund] #To refund the cash if desired
		custom_tooltip = rinta_refund_investment_tt
		hidden_effect = {
			if = {
				limit = { has_country_flag = rinta_telescope_triggered_amount }
				add_treasury = $amount$
			}
			else = {
				var_effect = {
					var = sky_incident_treasury_country_temp
					effect = add_treasury
					sign = +
				}
			}
			clr_country_flag = rinta_telescope_triggered_amount
		}
	]
	rinta_check_telescope_invest_level = yes
}

rinta_steal_funds_event_effect = {
	if = {
		limit = { has_global_flag = rinta_supercharge_stealing_funds_flag }

			who = event_target:$event_target$
			value = -20
		}
	}
	else = {
		add_spy_network_in = {
			who = event_target:$event_target$
			value = -35
		}
	}
	rinta_invest_in_telescope = { level = 0_5 }
	event_target:$event_target$ = {
		tooltip = {
			add_opinion = {
				who = ROOT
				modifier = rinta_stole_astronomy_funds

			add_casus_belli = {
				type = cb_spy_discovered

				months = 60 #5 years
			}

			id = rinta_seeker_events.520 #Astronomical Center Funds Stolen
		}
	}
	set_country_flag = stole_telescope_funds_flag #Reset decision cooldown
}
