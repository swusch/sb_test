	}

	}
	on_revoked = {

	on_granted = {

			estate_privilege = estate_mages_organization_acolytes
		}
	}
	penalties = {
	}
	benefits = {
	}
	on_invalid = {
		custom_tooltip = estate_mages_religious_invalid_tooltip
		hidden_effect = {
			if = {
				limit = { NOT = { has_mages_organization_privilege = yes } }
				set_estate_privilege = estate_mages_organization_state
			}
			remove_estate_privilege = estate_acolytes_land_share_10
			remove_estate_privilege = estate_acolytes_land_share_20
			remove_estate_privilege = estate_acolytes_land_share_30
			remove_estate_privilege = estate_acolytes_land_share_40
			remove_estate_privilege = estate_acolytes_land_share_50
			remove_estate_privilege = estate_acolytes_dominions_law_army_decentralisation
			remove_estate_privilege = estate_acolytes_dominions_law_army_centralisation
			remove_estate_privilege = estate_acolytes_dominions_law_the_dark_tithe
			remove_estate_privilege = estate_acolytes_dominions_law_centralized_banking
			remove_estate_privilege = estate_acolytes_dominions_law_power_split
			remove_estate_privilege = estate_acolytes_dominions_law_ibevaries_lab_rats
			remove_estate_privilege = estate_acolytes_dominions_law_gnomish_brain
			remove_estate_privilege = estate_acolytes_dominions_law_kobold_slaves
			remove_estate_privilege = estate_acolytes_dominions_law_warden_of_the_state_mage_minions
			remove_estate_privilege = estate_acolytes_dominions_law_warden_of_the_state_children_of_war
			remove_estate_privilege = estate_acolytes_dominions_law_warden_of_the_state_envoy_of_heart
			remove_estate_privilege = estate_acolytes_dominions_law_absolute_power
			remove_estate_privilege = estate_acolytes_dominions_law_timed_research
			remove_estate_privilege = estate_acolytes_dominions_law_sanguine_knowledge
		}
	}

	influence_scaled_conditional_modifier = {
		trigger = {
			NOT = {
				estate_loyalty = {
					estate = estate_mages
					loyalty = 30
				}
			}
		}
		modifier = {
			land_morale = -0.2
			yearly_absolutism = -0.2
			devotion = -2
			reduced_liberty_desire = -50
		}
	}
	influence_scaled_conditional_modifier = {	#Modifier that are applied if the specified conditions are true. At 100% Influence you gain 100% of the modifier.
		trigger = {
			estate_loyalty = {
				estate = estate_mages
				loyalty = 30
			}
			NOT = {
				estate_loyalty = {
					estate = estate_mages

				}
			}
		}
