	name = "racial_mil_change_racial_military_1"
	potential = { always = yes } #Should be beneath the other buttons
	trigger = {	}
	effect = {
		# Get custom tooltip
		racial_mil_effect_selector_dlc = { effect = racial_mil_cannot_change_to_selected_race_military }
		# Do nothing
		# Refresh ui
		hidden_effect = {
			racial_pop_interface_update_main = yes # Generic refresh
		}
	}
	tooltip = racial_mil_switch_button_tt
}



custom_button = { # Can switch racial military
	name = "racial_mil_change_racial_military_2"
	potential = {
		mil_power = 200
		racial_mil_can_switch_to_selected_race_military_trigger = yes
	}
	trigger = {
		#Putting these conditions here as a trigger to disable it if they wouldn't be able to open the old mil event
		#and inform them why they can't. They should still be able to access the info and whether they could otherwise switch
		if = {
			limit = { has_country_modifier = restructuring_military }
			custom_trigger_tooltip = {
				tooltip = racial_mil_is_restructuring_military_cttt
				NOT = { has_country_modifier = restructuring_military }
			}
		}
		# Has generated tt
		if = {
			limit = { has_government_attribute = disables_swap_racial_mil }
			NOT = { has_government_attribute = disables_swap_racial_mil }
		}
		if = {
			limit = { has_country_flag = disable_racial_mil_swap_flag }
			custom_trigger_tooltip = {
				tooltip = racial_mil_cannot_switch_gov_attribute_tt
				NOT = { has_country_flag = disable_racial_mil_swap_flag }
			}
		}
		
	}
	effect = {
		racial_mil_effect_selector_dlc = { effect = racial_mil_change_military_effect }
		racial_mil_change_mil_sub_effects = yes
		hidden_effect = {
			racial_pop_interface_update_main = yes # Generic refresh
		}
	}
	tooltip = racial_mil_switch_button_tt
}
