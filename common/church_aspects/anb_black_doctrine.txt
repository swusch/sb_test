
absolute_servitude = {
	cost = 100
	trigger = {
		is_ruler_enchantment_level_1_minimum = yes
		NOT = { has_country_modifier = bd_edict_of_absolute_servitude }
	}
	ai_will_do = {
		factor = 10
		modifier = {
			factor = 5
			rebel_progress_at_least = { value = 50 }
		}
		modifier = {
			factor = 3
			average_autonomy_above_min = 10
		}
	}
	effect = {
		if = {
			limit = {
				absolutism = 80
				is_ruler_enchantment_level_3 = yes
			}
			add_rebel_progress = { value = -40 }
		}
		else_if = {
			limit = {
				absolutism = 40
				is_ruler_enchantment_level_2_minimum = yes
			}
			add_rebel_progress = { value = -20 }
		}
		else = {
			add_rebel_progress = { value = -10 }
		}

		custom_tooltip = bd_absolute_servitude_effect_tt 
		hidden_effect = {
			add_country_modifier = {
				name = bd_edict_of_absolute_servitude
				duration = 3650
			}
		}
	}
	modifier = {}
}
