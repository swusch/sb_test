fangaulan_pantheon_aro_aspect = {
	sprite = "GFX_ancestral_armaments"
	cost = 200

	potential = {
	}

	trigger = {
		fangaulan_pantheon_holy_site_owner_check = {
			province_id = 390 #Sirik
		}
		custom_trigger_tooltip = {
			tooltip = fangaulan_pantheon_no_active_mod_tt
			NOT = {
				has_country_modifier = fangaulan_pantheon_aro_craft
				has_country_modifier = fangaulan_pantheon_aro_ten_task
			}
		}  	
	}
	
	effect = {
		country_event = { id = fangaulan_pantheon.1001 }
	}
	
	ai_will_do = {
		factor = 1
		modifier = {
			factor = 1.5
			OR = {
				AND = {
					NOT = { has_country_modifier = fangaulan_pantheon_aro_craft }
					treasury = 500
				}
				AND = {
					NOT = { has_country_modifier = fangaulan_pantheon_aro_ten_task }
					OR = {
						NOT = { adm = 3 }
						NOT = { dip = 3 }
						NOT = { mil = 3 }
					}
				}
			}
		}
	}
}
