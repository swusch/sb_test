		localisation_key = rpPopPolicyFocusExpel

	}
	text = {
		localisation_key = rpPopPolicyPurge

		}
	}

		localisation_key = rpPopPolicyFocusPurge

			has_country_modifier = racial_pop_centaur_purge

		}
	}
	text = {
		localisation_key = rpPopPolicyFocus
		trigger = { has_country_modifier = racial_focus_centaur }
